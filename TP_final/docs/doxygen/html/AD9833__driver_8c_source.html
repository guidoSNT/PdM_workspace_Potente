51 if (device.init == false) return AD_NOT_INIT;
52
53 ad9833_st_t res;
54 uint16_t reg_temp;

56 switch (form) {

81ad9833_st_t ad9833_set_phase(uint16_t phase, reg_t reg) {
82 if (device.init == false) return AD_NOT_INIT;
83
84 if (phase > AD9833_PHASE_WORD_MASK) return AD_BAD_ARGUMENT;
85
86 uint16_t ctrl_temp = device.ctrl_reg & ~AD9833_CTRL_B28;
87 ctrl_temp = (reg == REG_0) ? ctrl_temp | AD9833_CTRL_FSELECT : ctrl_temp & ~AD9833_CTRL_FSELECT;
88 uint16_t phase_reg = ((uint32_t) phase << AD9833_PHASE_REG_LEN) / (2 * AD9833_PI) - AD9833_CONVERSION_OFFSET;
89 uint16_t reg_temp = (reg == REG_0) ? AD9833_REG_PHASE0 : AD9833_REG_PHASE1;
90 phase_reg &= AD9833_PHASE_WORD_MASK;
91
92 if (ad9833_set_control(ctrl_temp) != AD_OK) return AD_NO_RESPONSE;
93
94 if (AD_SPI_OK != write_word(reg_temp | phase_reg)) return AD_FAIL;
95
96 if (reg == REG_0) device.phase0 = phase_reg;
97 else device.phase1 = phase_reg;
98
99 return AD_OK;

102ad9833_st_t ad9833_set_freq(uint32_t freq, reg_t reg) {
103 if (device.init == false) return AD_NOT_INIT;
104
105 // Max freq is the nyquist frec at fs/2
106 if (freq > device.fmclk * AD9833_MHZ_IN_dHZ / 2) return AD_BAD_ARGUMENT;
107
108 // To properly set the freq, we need to check which register (reg_temp) and
109 // get the 28-bit register. curr_freq is to check new and old register.
110 uint32_t freq_reg = get_freq_reg(freq);
111 uint32_t curr_freq = (reg == REG_0) ? device.freq0 : device.freq1;
112 uint16_t reg_temp = (reg == REG_0) ? AD9833_REG_FREQ0 : AD9833_REG_FREQ1;
113
114 // When both LSB and MSB need to be set then B28 bit has to be set for 2
115 // consecutive comms (first is LSB and next is MSB). This is tested by looking
116 // if the 14 MSB are the same and if the freq_reg has more than 14 bits
117 bool full_tx = GET_14_MSB(freq_reg) != GET_14_MSB(curr_freq);
118 uint16_t ctrl_temp = (full_tx) ? device.ctrl_reg | AD9833_CTRL_B28 : device.ctrl_reg & ~AD9833_CTRL_B28;
119 ctrl_temp = (reg == REG_0) ? ctrl_temp | AD9833_CTRL_FSELECT : ctrl_temp & ~AD9833_CTRL_FSELECT;
120
121 if (ad9833_set_control(ctrl_temp) != AD_OK) return AD_NO_RESPONSE;
122
123 if (AD_SPI_OK != write_word(reg_temp | (uint16_t) (freq & AD9833_FREQ_14BIT_MASK))) return AD_FAIL;
124
125 if (full_tx) {
126 if (AD_SPI_OK != write_word(reg_temp | (uint16_t) ((freq >> AD9833_FREQ_14BIT_SHIFT) & AD9833_FREQ_14BIT_MASK))) return AD_FAIL;
127 }
128
129 if (reg == REG_0) device.freq0 = freq_reg;
130 else device.freq1 = freq_reg;
131
132 return AD_OK;
133}
