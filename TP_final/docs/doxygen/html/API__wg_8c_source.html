110void wg_fsm_init() {
111 wg_st = WG_IDLE;
112
113 // Init all the buttons
114 for (uint8_t i = 0; i < AMOUNT_BUTTONS; i++) {
115 inputs[i].btn_id = i;
116 debounceFSM_init(&inputs[i]);
117 }
118
119 // Init both interfaces
120 uartInit();
121 adc_port_init();
122
123 // Setup the ad9833
124 if (ad9833_init(INIT_FMCLK_MHZ) != AD_OK) return;
125
126 if (ad9833_set_freq(INIT_FREQ_DHZ, REG_0) != AD_OK) return;
127
128 if (ad9833_set_phase(INIT_PHASE_DHZ, REG_0) != AD_OK) return;
129
130 if (ad9833_set_form(SINE) != AD_OK) return;
131
132 // Clearing the screen allows that the string isn't imposed in the rest of the terminal
133 uartSendStringSize((uint8_t *) ANSI_CLEAR, sizeof(ANSI_CLEAR));
134 memset(freq_str, EOL, sizeof(freq_str));
135
136 freq_to_digit(INIT_FREQ_DHZ);
137 curr_phase = INIT_PHASE_DHZ;
138 curr_form = SINE;
139 initialized = true;
140 btns = 0;
141}

143bool wg_fsm_update() {
144 char buff[FSM_BUFFER_LEN];
145 if (initialized != true) return false;
146
147 switch (wg_st) {
148 case WG_IDLE:
149 input_getter();
150 if (btns != NO_PRESSED_BTNS) wg_st = WG_PRC_IN;
151 break;
152
153 case WG_PRC_IN:
154 wg_st = (check_bits()) ? WG_EXEC : WG_ERROR;
155 break;
156
157 case WG_EXEC:
158 wg_st = (process_input()) ? WG_CNG_UART : WG_ERROR;
159 break;
160
161 case WG_CNG_UART:
162 memset(buff, EOL, sizeof(buff));
163 if (wg_format_display(curr_form, curr_phase, curr_pos, buff, sizeof(buff)) == 0) {
164 wg_st = WG_ERROR;
165 break;
166 }
167
168 wg_st = WG_IDLE;
169 uartSendStringSize((uint8_t *) buff, strnlen(buff, sizeof(buff)));
170 break;
171
172 case WG_ERROR:
173 uartSendStringSize((uint8_t *) "ERROR\r\n", strlen("ERROR\r\n"));
174 wg_st = WG_IDLE;
175 break;
176 }
177 return true;
178}

182// The idea was to reduce the amount of snprintf (ideally it shouldn't use any), thas why the usage of man strlcat
183static uint16_t wg_format_display(ad9833_wf_t form, uint16_t phase, uint8_t pos, char *buff, uint16_t buff_len) {
184 if (buff == NULL || buff_len == 0 || pos > MAX_DIGITS_FREQ) return 0;
185
186 // Get the freq string
187 uint16_t freq_len = strnlen(freq_str, MAX_DIGITS_FREQ) - ONE_OFFSET;
188 uint8_t blink_idx = (pos == 0) ? UINT8_MAX : freq_len - pos; // to point the correct char
189 char *p = buff;
190 const char *end = buff + buff_len - ONE_OFFSET;
191 const char *wf_str = ((uint8_t) form < WF_NAMES_COUNT && wf_names[form] != NULL) ? wf_names[form] : "UNKNOWN";
192
193 // To remove anything from the terminal, go to the beginning and remove the cursor, we add the following commands at the start
194 strlcat(p, ANSI_CLEAR ANSI_HOME ANSI_HIDE_CURSOR, end - p);
195 if (p + strlen(ANSI_CLEAR ANSI_HOME ANSI_HIDE_CURSOR) >= end) return 0;
196 p += strlen(ANSI_CLEAR ANSI_HOME ANSI_HIDE_CURSOR);
197
198 // Find the digit and add the highlight
199 for (uint16_t i = 0; i < freq_len; i++) {
200 if (i == blink_idx && blink_idx >= 0) {
201 int16_t written = snprintf(p, (size_t) (end - p), ANSI_HIGHLIGHT "%c" ANSI_HIGHLIGHT_OFF, freq_str[i]);
202 if (written < 0 || p + written >= end) return 0;
203 p += written;
204 } else {
205 if (p >= end) return 0;
206 *p++ = freq_str[i];
207 }
208 }
209
210 // Decimal point
211 if (p >= end) return 0;
212 *p++ = DOT_CHAR;
213
214 // Decimal digit
215 if (pos == 0) {
216 int16_t written = snprintf(p, (size_t) (end - p), ANSI_HIGHLIGHT "%c" ANSI_HIGHLIGHT_OFF, freq_str[MAX_DIGITS_FREQ - ONE_OFFSET]);
217 if (written < 0 || p + written >= end) return 0;
218 p += written;
219 } else {
220 if (p >= end) return 0;
221 *p++ = freq_str[MAX_DIGITS_FREQ - ONE_OFFSET];
222 }
223
224 // Add end of freq
225 strlcat(p, END_FREQ_STR, end - p);
226 if (p + strlen(END_FREQ_STR) >= end) return 0;
227 p += strlen(END_FREQ_STR);
228
229 // Print the phase
230 int16_t written = snprintf(p, (size_t) (end - p), "%u.%03u rads\r\n", phase / AD9833_FREQ_SCALE_MRAD_RAD, phase % AD9833_FREQ_SCALE_MRAD_RAD);
231 if (written < 0 || p + written >= end) return 0;
232 p += written;
233
234 // Append the wave form
235 strlcat(p, wf_str, end - p);
236 if (p + strlen(wf_str) >= end) return 0;
237 p += strlen(wf_str);
238
239 // Append the Reset
240 strlcat(p, ANSI_RESET, end - p);
241 if (p + strlen(ANSI_RESET) >= end) return 0;
242 p += strlen(ANSI_RESET);
243 *p = EOL;
244
245 return (uint16_t) (p - buff);
246}

248static void input_getter() {
249 uint8_t state = 0;
250 uint16_t adc_val;
251 btns = 0;
252
253 if (adc_port_read(&adc_val)) {
254 uint16_t new_phase = ((uint32_t) adc_val * AD9833_MAX_PHASE_MRADS) / ADC_RESOLUTION;
255 uint16_t diff = (new_phase > curr_phase) ? (new_phase - curr_phase) : (curr_phase - new_phase);
256 if (diff > ADC_PHASE_THRESHOLD) {
257 curr_phase = new_phase;
258 state = BTN_PHASE;
259 }
260 }
261
262 for (uint8_t i = 0; i < AMOUNT_BUTTONS; i++) {
263 debounceFSM_update(&inputs[i]);
264 state |= (uint8_t) readKey(&inputs[i]) << i;
265 }
266 btns = state;
267}

269static bool process_input() {
270 switch (btns) {
271 case BTN_LOAD:
272 ad9833_set_phase(curr_phase, REG_0);
273 ad9833_set_freq(digit_to_freq(), REG_0);
274 ad9833_set_form(curr_form);
275 break;
276
277 case BTN_LEFT:
278 curr_pos++;
279 curr_pos %= MAX_DIGITS_FREQ;
280 break;
281
282 case BTN_RIGHT:
283 curr_pos = (curr_pos == 0) ? MAX_DIGITS_FREQ - ONE_OFFSET : curr_pos - ONE_OFFSET;
284 break;
285
286 case BTN_UP:
287 increase_digit(curr_pos);
288 break;
289
290 case BTN_DOWN:
291 reduce_digit(curr_pos);
292 break;
293
294 case BTN_CHANGE:
295 curr_form++;
296 curr_form %= SQUARE_DIV + ONE_OFFSET;
297 break;
298
299 case BTN_PHASE:
300 break;
301
302 default:
303 btns = 0; // Reset the btns for next read
304 return false;
305 }
306 btns = 0;
307 return true;
308}

310void freq_to_digit(uint32_t freq) {
311 for (uint8_t i = 0; i < MAX_DIGITS_FREQ; i++) {
312 freq_str[MAX_DIGITS_FREQ - ONE_OFFSET - i] = (freq % DECIMAL_BASE) + ZERO_CHAR;
313 freq /= DECIMAL_BASE;
314 }
315 freq_str[MAX_DIGITS_FREQ] = EOL;
316}

318uint32_t digit_to_freq(void) {
319 uint32_t val = 0;
320 for (int i = 0; i < MAX_DIGITS_FREQ; i++) { val = (val * DECIMAL_BASE) + (freq_str[i] - ZERO_CHAR); }
321 return val;
322}

324void increase_digit(uint8_t pos) {
325 while (pos < MAX_DIGITS_FREQ) {
326 int idx = MAX_DIGITS_FREQ - ONE_OFFSET - pos;
327
328 if (freq_str[idx] == NINE_CHAR) {
329 freq_str[idx] = ZERO_CHAR;
330 pos++;
331 } else {
332 freq_str[idx]++;
333 break;
334 }
335 }
336}

338void reduce_digit(uint8_t pos) {
339 bool can_borrow = false;
340 for (int i = pos; i < MAX_DIGITS_FREQ; i++) {
341 if (freq_str[MAX_DIGITS_FREQ - ONE_OFFSET - i] > ZERO_CHAR) {
342 can_borrow = true;
343 break;
344 }
345 }
346
347 // Already 0
348 if (!can_borrow) { return; }
349
350 while (pos < MAX_DIGITS_FREQ) {
351 int idx = MAX_DIGITS_FREQ - ONE_OFFSET - pos;
352
353 if (freq_str[idx] == ZERO_CHAR) {
354 freq_str[idx] = NINE_CHAR;
355 pos++;
356 } else {
357 freq_str[idx]--;
358 break;
359 }
360 }
