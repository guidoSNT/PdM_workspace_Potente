8bool_t uartInit() {
9 // Config string (it's weird and has to be configured if uart_handle is
10 // changed but it's better than using an printf with the raw values)
11 char config[] =
12 API_UART_INSTANCE_STR "|" MACRO_STR(API_UART_BUADRATE) "|" API_UART_WL_STR "|" API_UART_SB_STR "|" API_UART_PARITY_STR "|" API_UART_MODE_STR "|" API_UART_HWCTL_STR "|" API_UART_OS_STR "\r\n";
13
14 uart_handle = (UART_HandleTypeDef) {
15 .Instance = API_UART_INSTANCE,
16 .Init.BaudRate = API_UART_BUADRATE,
17 .Init.WordLength = API_UART_WL,
18 .Init.StopBits = API_UART_SB,
19 .Init.Parity = API_UART_PARITY,
20 .Init.Mode = API_UART_MODE,
21 .Init.HwFlowCtl = API_UART_HWCTL,
22 .Init.OverSampling = API_UART_OS,
23 };
24
25 // Init the UART and send the config on success
26 if (HAL_UART_Init(&uart_handle) != HAL_OK) return false;
27
28 HAL_UART_Transmit(&uart_handle, (uint8_t *) config, strlen(config), HAL_MAX_DELAY);
29 uart_initialized = true;
30 return true;
31}

33void uartSendString(uint8_t *pstring) {
34 size_t str_size = 0;
35 char *str_cp = (char *) pstring;
36
37 // Check for NULL pointer
38 if (pstring == NULL) return;
39
40 // If str_cp doesn't end with \0 this will read out of bounds.
41 // Using this instead of strlen() is to make clear the issue without the \0.
42 while (*str_cp++ != '\0' && str_size < API_UART_MAX_SIZE) str_size++;
43
44 HAL_UART_Transmit(&uart_handle, pstring, str_size, HAL_MAX_DELAY);
45}

47void uartSendStringSize(uint8_t *pstring, uint16_t size) {
48 // This should be separated to use an error handler

50
51 // check if size = actual size (also checks if null-terminated)
52 if (strnlen((char *) pstring, size) != size) return;
53
54 HAL_UART_Transmit(&uart_handle, pstring, size, HAL_MAX_DELAY);
